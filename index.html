function doLogin(){ var btn=document.getElementById('loginBtn'); var email=document.getElementById('email').value.trim(); var pass=document.getElementById('pass').value.trim(); hideAlert('loginAlert'); if(!email||!pass){ showAlert('loginAlert','loginAlertTxt','Please enter both email and password.','error'); return; } btn.disabled=true; btn.innerHTML=' Signing in…'; setLoading(true,'Verifying credentials…'); google.script.run .withSuccessHandler(function(res){ setLoading(false); btn.disabled=false; btn.innerHTML=' Sign In'; if(res && res.success){ if(document.getElementById('rememberMe').checked){ window.parent.postMessage({type:'SAVE_LOGIN', email:email, pass:pass}, '*'); } else { window.parent.postMessage({type:'CLEAR_LOGIN'}, '*'); } document.getElementById('sName').value = res.fullName || ''; document.getElementById('sEmail').value = res.email || ''; document.getElementById('sProject').value = res.project || ''; document.getElementById('sDesignation').value = res.designation || ''; document.getElementById('sLocation').value = res.location || ''; document.getElementById('sJoining').value = res.joiningDate || ''; document.getElementById('sPhoto').value = res.photo || ''; document.getElementById('loginPanel').classList.add('hidden'); if(res.isHR){ document.getElementById('hrUserTag').textContent = res.fullName || 'HR Admin'; document.getElementById('hrDashboard').classList.remove('hidden'); loadPendingRequests(); } else { document.getElementById('staffDashboard').classList.remove('hidden'); loadMyRequestsSilent(document.getElementById('sEmail').value||res.email||''); var el; el = document.getElementById('profileName'); if(el) el.textContent = res.fullName || ''; el = document.getElementById('profileProject'); if(el) el.textContent = res.project || ''; el = document.getElementById('profileLocation'); if(el) el.textContent = res.location || ''; el = document.getElementById('profileDesignation'); if(el) el.textContent = res.designation || ''; el = document.getElementById('profileJoining'); if(el) el.textContent = res.joiningDate || ''; el = document.getElementById('profilePhoto'); if(el){ el.src = res.photo ? res.photo : "https://ui-avatars.com/api/?name=" + encodeURIComponent(res.fullName || 'User'); } switchStaffView('menu'); } toast('Welcome back!',res.fullName||'','t-ok'); } else { showAlert('loginAlert','loginAlertTxt',(res&&res.error)||'Login failed.','error'); } }) .withFailureHandler(function(err){ setLoading(false); btn.disabled=false; btn.innerHTML=' Sign In'; showAlert('loginAlert','loginAlertTxt','Connection error: '+err,'error'); }) .handleLogin(email,pass); }
